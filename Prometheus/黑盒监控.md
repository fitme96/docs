
探测端增加如下服务
```yaml

  blackbox-exporter:
    image: prom/blackbox-exporter
    command:
      - '--config.file=/config/blackbox.yml'
    volumes:
      - './blackbox.yml:/config/blackbox.yml'
```

blackbox.yml 定义模块
```yml
cat blackbox.yml

modules:
  http_2xx:
    prober: http
    http:
      method: GET
  http_post_2xx:
    prober: http
    http:
      method: POST
  http_ingore_ca:  #这里增加了一个忽略ssl证书的url探测模块
        prober: http
        http:
          method: GET
          preferred_ip_protocol: "ip4"
          tls_config:
            insecure_skip_verify: true
```

http状态码告警规则和ssl过期告警规则
```
cat rules/blackbox.rules 
groups:
- name: httpStatus
  rules:
  - alert: httpStatus
    expr: probe_http_status_code{env="sec"} != 200 # 通过标签过滤不检测站点
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "实例 {{ $labels.instance }} 不等于200"
      description: "{{ $labels.instance }} 状态码为  {{ $value }}"
  - alert: CertExpiry
    expr: probe_ssl_earliest_cert_expiry{env="ssl"} - time() < 86400 * 30 通过标签过滤非ssl站点
    for: 1m
    labels:
      severity: warning
    annotations:
      summary:  域名证书即将过期 (instance {{ $labels.instance }})
      description: "域名证书30天后过期"
```
